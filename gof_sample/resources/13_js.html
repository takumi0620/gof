/*
 *  HelpTopics
 *
 *  ヘルプ項目を表す定数を定義する
 */
public class HelpTopics {
    public static final int NO_HELP_TOPIC = -1;
    public static final int PRINT_TOPIC = 1;
    public static final int PAPER_ORIENTATION_TOPIC = 2;
    public static final int APPLICATION_TOPIC = 3;
}


/*
 *  HelpHandler
 *
 *  役割：<Handler>
 */
public abstract class HelpHandler {
    public HelpHandler(HelpHandler handler, int topic) {
        _successor = handler;
        _topic = topic;
    }
    public boolean hasHelp() {
        return _topic != HelpTopics.NO_HELP_TOPIC;
    }
    public void setHandler(HelpHandler handler, int topic) {
        _successor = handler;
        _topic = topic;
    }
    public void handleHelp() {
        if (_successor != null) {
            _successor.handleHelp();
        }
    }

    private HelpHandler _successor;
    private int _topic;
}


/*
 *  Widget
 *
 *  役割：全ての GUI 部品が継承する抽象クラス
 */
public abstract class Widget extends HelpHandler {
    public Widget(Widget parent, int topic) {
        super(parent, topic);
        _parent = parent;
    }

    private Widget _parent;
}


/*
 *  Button
 *
 *  役割：<ConcreteHandler>
 */
public class Button extends Widget {
    public Button(Widget parent, int topic) {
        super(parent, topic);
    }

    public void handleHelp() {
        if (hasHelp()) {
            // ヘルプメッセージを表示する
        } else {
            super.handleHelp();
        }
    }
    // その他 Button がオーバーライドする Widget のメソッド...
}


/*
 *  Dialog
 *
 *  役割：<ConcreteHandler>
 */
public class Dialog extends Widget {
    public Dialog(HelpHandler handler, int topic) {
        super(null, HelpTopics.NO_HELP_TOPIC);
        setHandler(handler, topic);
    }

    public void handleHelp() {
        if (hasHelp()) {
            // ヘルプメッセージを表示する
        } else {
            super.handleHelp();
        }
    }
    // その他 Button がオーバーライドする Widget のメソッド...
}


/*
 *  Application
 *
 *  役割：<ConcreteHandler>
 */
public class Application extends HelpHandler {
    public Application(int topic) {
        super(null, topic);
    }

    public void handleHelp() {
        // ヘルプ項目の一覧を表示
    }

    // アプリケーション固有の処理...
}
