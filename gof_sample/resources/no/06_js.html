Java 版サンプルコードでは「オブジェクトアダプタ」例を示す。Java で はクラスの多重継承が許されていないため、一般にクラスアダプタは実装 できない。 


import java.awt.Point;
import java.awt.Dimension;
import java.awt.Rectangle;

/*
 *  Shape
 *
 *  役割：<Target>
 */
public abstract class Shape {

    /*
     *  <Target> インターフェース
     */

    public abstract Rectangle boundingBox();
    public abstract Manipulator createManipulator();
}


/*
 *  TextView
 *
 *  役割：<Adaptee>
 */
public class TextView {

    /*
     *  コンストラクタ
     */

    public TextView() {
        // ...
    }


    /*
     *  <Adaptee> インターフェース
     */

    // <Target> インターフェースへの変換の対象となる部分

    public Point getOrigin() {
        // ...
    }
    public Dimension getExtent() {
        // ...
    }

    // <Target> 中に対応するインターフェースがない部分

    public boolean isEmpty() {
        // ...
    }


    // ...
}


/*
 *  TextShape
 *
 *  役割：<Adapter>
 */
public class TextShape extends Shape {

    /*
     *  コンストラクタ
     */

    public TextShape(TextView t) {
        _text = t;
    }


    /*
     *  <Target> インターフェース
     */

    public Rectangle boundingBox() {
        Point origin;
        Dimension extent;

        origin = _text.getOrigin();
        extent = _text.getExtent();

        return new Rectangle(origin, extent);
    }

    public Manipulator createManipulator() {
        return new TextManipulator(this);
    }

    /*
     *  <Adaptee> インターフェースの一部
     */

    // TextManipulator が使用するメソッド

    public boolean isEmpty() {
        return _text.isEmpty();
    }


    /*
     *  プライベートメンバ
     */

    private TextView _text; // <Adaptee>
}
