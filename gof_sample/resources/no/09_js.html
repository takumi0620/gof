/*
 *  Decorator
 *
 * 役割：<Decorator>
 */
public abstract class Decorator extends VisualComponent {
    Decorator(VisualComponent vc) {
        _component = vc;
    }

    public void draw() {
        _component.draw();
    }

    public void resize() {
        _component.resize();
    }

    private VisualComponent _component;
}
