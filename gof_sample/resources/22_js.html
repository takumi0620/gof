/*
 *  View
 *
 *  役割：<AbstractClass>
 */
public abstract class View {
    // ...

    public void setFocus() {
        // ...
    }
    public void resetFocus() {
        // ...
    }

    /*
     *  テンプレートメソッド
     */
    public void display() {
        setFocus();
        doDisplay();
        resetFocus();
    }

    /*
     *  プリミティブオペレーション
     */
    public abstract void doDisplay();
}


/*
 *  MyView
 *
 *  役割：<ConcreteClass>
 */
public class MyView extends View {
    // ...

    /*
     *  プリミティブオペレーション
     */
    public void doDisplay() {
        // 内容を描画する
    }
}
